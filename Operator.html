Complex Number Class

Implement arithmetic operators for complex numbers:

complex.cpp 
#include <iostream>
using namespace std;

class Complex {
private:
    double real, imag;

public:
    // Constructor
    Complex(double r = 0, double i = 0) : real(r), imag(i) {}
    
    // Overload + operator
    Complex operator+(const Complex& other) const {
        return Complex(real + other.real, imag + other.imag);
    }
    
    // Overload - operator
    Complex operator-(const Complex& other) const {
        return Complex(real - other.real, imag - other.imag);
    }
    
    // Overload * operator
    Complex operator*(const Complex& other) const {
        return Complex(
            real * other.real - imag * other.imag,
            real * other.imag + imag * other.real
        );
    }
    
    // Overload == operator
    bool operator==(const Complex& other) const {
        return real == other.real && imag == other.imag;
    }
    
    // Display method
    void print() const {
        cout << real;
        if (imag >= 0) cout << " + " << imag << "i";
        else cout << " - " << -imag << "i";
    }
};

int main() {
    Complex a(3, 4);
    Complex b(1, 2);
    
    cout << "a = "; a.print(); cout << endl;
    cout << "b = "; b.print(); cout << endl;
    
    Complex c = a + b;
    cout << "a + b = "; c.print(); cout << endl;
    
    Complex d = a * b;
    cout << "a * b = "; d.print(); cout << endl;
    
    cout << "a == b: " << (a == b ? "true" : "false") << endl;
    
    return 0;
}

Prefix vs Postfix Increment

++x PREFIX

Increment First

  • No dummy parameter
  • Returns reference to *this
  • More efficient (no copy)
x++ POSTFIX

Return Then Increment

  • Dummy int parameter
  • Returns copy of old value
  • Creates temporary object
increment.cpp 
#include <iostream>
using namespace std;

class Counter {
private:
    int val;

public:
    Counter(int v = 0) : val(v) {}
    
    int get() const { return val; }
    
    // PREFIX: ++counter
    Counter& operator++() {
        ++val;
        return *this;
    }
    
    // POSTFIX: counter++ (dummy int)
    Counter operator++(int) {
        Counter old = *this;
        ++val;
        return old;
    }
};

int main() {
    Counter c(5);
    
    cout << "Start: " << c.get() << endl;
    
    cout << "++c: " << (++c).get() << endl;  // 6
    cout << "After: " << c.get() << endl;   // 6
    
    cout << "c++: " << (c++).get() << endl;  // 6 (old)
    cout << "After: " << c.get() << endl;   // 7
    
    return 0;
}

Subscript Operator []

Create safe array access with bounds checking:

safe_array.cpp 
#include <iostream>
using namespace std;

class SafeArray {
private:
    int* data;
    int size;

public:
    SafeArray(int s) : size(s) {
        data = new int[size]();
    }
    
    ~SafeArray() { delete[] data; }
    
    // Subscript operator for read/write
    int& operator[](int idx) {
        if (idx < 0 || idx >= size) {
            cout << "[ERROR] Index " << idx << " out of bounds!" << endl;
            return data[0];
        }
        return data[idx];
    }
    
    int len() const { return size; }
};

int main() {
    SafeArray arr(5);
    
    for (int i = 0; i < 5; i++) {
        arr[i] = (i + 1) * 10;
    }
    
    cout << "Array: ";
    for (int i = 0; i < arr.len(); i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    // Test bounds check
    arr[10] = 99;
    
    return 0;
}

Friend Function Example

friend_func.cpp 
#include <iostream>
using namespace std;

class Box {
private:
    double width, height;

public:
    Box(double w, double h) : width(w), height(h) {}
    
    // Declare friend - can access private
    friend double getArea(const Box& b);
    friend ostream& operator<<(ostream& os, const Box& b);
};

// Friend function (NOT a member)
double getArea(const Box& b) {
    return b.width * b.height;  // Direct private access!
}

// Friend operator<<
ostream& operator<<(ostream& os, const Box& b) {
    os << "Box[" << b.width << "x" << b.height << "]";
    return os;
}

int main() {
    Box b1(5, 4);
    Box b2(3, 6);
    
    cout << b1 << " Area: " << getArea(b1) << endl;
    cout << b2 << " Area: " << getArea(b2) << endl;
    
    return 0;
}

Friend Class Example

friend_class.cpp 
#include <iostream>
using namespace std;

class Engine;

class Car {
private:
    string model;
    int speed;
    bool running;

public:
    Car(string m) : model(m), speed(0), running(false) {}
    
    friend class Engine;  // Engine can access all
    
    void status() {
        cout << model << ": " << (running ? "ON" : "OFF") 
             << ", " << speed << " mph" << endl;
    }
};

class Engine {
public:
    void start(Car& c) {
        c.running = true;
        cout << "Started: " << c.model << endl;
    }
    
    void accelerate(Car& c, int amt) {
        if (c.running) c.speed += amt;
    }
    
    void stop(Car& c) {
        c.running = false;
        c.speed = 0;
    }
};

int main() {
    Car car("Tesla");
    Engine eng;
    
    car.status();
    eng.start(car);
    eng.accelerate(car, 60);
    car.status();
    eng.stop(car);
    car.status();
    
    return 0;
}

Pointer Basics

ptr_basics.cpp 
#include <iostream>
using namespace std;

int main() {
    int x = 42;
    int* ptr = &x;  // ptr holds address of x
    
    cout << "x = " << x << endl;
    cout << "&x = " << &x << endl;
    cout << "ptr = " << ptr << endl;
    cout << "*ptr = " << *ptr << endl;
    
    // Modify through pointer
    *ptr = 100;
    cout << "After *ptr = 100: x = " << x << endl;
    
    // Null pointer
    int* np = nullptr;
    if (np == nullptr) {
        cout << "np is null" << endl;
    }
    
    // Pointer to pointer
    int** pp = &ptr;
    cout << "**pp = " << **pp << endl;
    
    return 0;
}

Pointer Arithmetic

ptr_math.cpp 
#include <iostream>
using namespace std;

int main() {
    int arr[] = {10, 20, 30, 40, 50};
    int* p = arr;  // Array decays to pointer
    
    cout << "Using pointer arithmetic:" << endl;
    for (int i = 0; i < 5; i++) {
        cout << "*(p+" << i << ") = " << *(p + i) << endl;
    }
    
    cout << "\nAddresses:" << endl;
    cout << "p   = " << p << endl;
    cout << "p+1 = " << (p+1) << endl;
    cout << "p+2 = " << (p+2) << endl;
    
    // Pointer difference
    int* start = &arr[0];
    int* end = &arr[4];
    cout << "\nDistance: " << (end - start) << " elements" << endl;
    
    return 0;
}
üí° Key Insight

ptr + 1 adds sizeof(type) bytes, not 1 byte. For int*, that's 4 bytes.

Dynamic Memory

dynamic_mem.cpp 
#include <iostream>
using namespace std;

class Student {
public:
    string name;
    Student(string n) : name(n) {
        cout << "Created: " << name << endl;
    }
    ~Student() {
        cout << "Destroyed: " << name << endl;
    }
};

int main() {
    // Single allocation
    int* num = new int(42);
    cout << "*num = " << *num << endl;
    delete num;
    
    // Array allocation
    int* arr = new int[5];
    for (int i = 0; i < 5; i++) arr[i] = i * 10;
    delete[] arr;  // Use delete[] for arrays!
    
    // Object allocation
    Student* s = new Student("Alice");
    cout << "Name: " << s->name << endl;
    delete s;
    
    return 0;
}

Function Pointers

func_ptr.cpp 
#include <iostream>
using namespace std;

int add(int a, int b) { return a + b; }
int sub(int a, int b) { return a - b; }
int mul(int a, int b) { return a * b; }

void calc(int a, int b, int (*op)(int, int), string name) {
    cout << name << "(" << a << "," << b << ") = " << op(a, b) << endl;
}

int main() {
    // Declare function pointer
    int (*fp)(int, int);
    
    fp = add;
    cout << "fp(5,3) = " << fp(5, 3) << endl;
    
    fp = mul;
    cout << "fp(5,3) = " << fp(5, 3) << endl;
    
    // Array of function pointers
    int (*ops[3])(int, int) = {add, sub, mul};
    string names[] = {"add", "sub", "mul"};
    
    for (int i = 0; i < 3; i++) {
        calc(10, 5, ops[i], names[i]);
    }
    
    return 0;
}
