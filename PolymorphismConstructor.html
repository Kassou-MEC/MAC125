Section 01

Understanding Polymorphism

"Many forms" ‚Äî the ability of objects to take on different behaviors while sharing a common interface

‚ö° Static Binding

Compile-Time Polymorphism

Resolved during compilation. The compiler determines which function to call based on the function signature.

  • Function Overloading
  • Operator Overloading
  • Templates (Generic Programming)
  • Faster execution (no runtime overhead)
  • Type-safe at compile time
üéØ Dynamic Binding

Runtime Polymorphism

Resolved during program execution. Uses virtual functions and the vtable mechanism for dynamic dispatch.

  • Virtual Functions
  • Function Overriding
  • Abstract Classes & Interfaces
  • Flexible & extensible design
  • Enables plugin architectures
Aspect Compile-Time Runtime
When Resolved During compilation During program execution
Mechanism Function signatures, templates Virtual table (vtable)
Keywords None (implicit) virtual, override
Performance ‚úÖ Faster (no lookup) ‚ö†Ô∏è Slight overhead (vtable lookup)
Flexibility Less flexible ‚úÖ Highly flexible

Compile-Time: Function Overloading

Multiple functions share the same name but have different parameter lists. The compiler selects the correct version based on the arguments provided.

function_overloading.cpp 
#include <iostream>
using namespace std;

class Calculator {
public:
    // Overloaded add() - different parameter types and counts
    int add(int a, int b) {
        cout << "int + int: ";
        return a + b;
    }
    
    double add(double a, double b) {
        cout << "double + double: ";
        return a + b;
    }
    
    int add(int a, int b, int c) {
        cout << "int + int + int: ";
        return a + b + c;
    }
    
    string add(string a, string b) {
        cout << "string + string: ";
        return a + b;
    }
};

int main() {
    Calculator calc;
    
    // Compiler chooses correct version at compile time
    cout << calc.add(5, 3) << endl;           // int version
    cout << calc.add(2.5, 3.7) << endl;       // double version
    cout << calc.add(1, 2, 3) << endl;        // three-int version
    cout << calc.add("Hello ", "World") << endl; // string version
    
    return 0;
}

Compile-Time: Operator Overloading

Define custom behavior for C++ operators when used with your classes. Makes user-defined types feel natural to use.

operator_overloading.cpp 
#include <iostream>
using namespace std;

class Vector2D {
private:
    double x, y;

public:
    Vector2D(double x = 0, double y = 0) : x(x), y(y) {}
    
    // Overload + operator
    Vector2D operator+(const Vector2D& other) const {
        return Vector2D(x + other.x, y + other.y);
    }
    
    // Overload - operator
    Vector2D operator-(const Vector2D& other) const {
        return Vector2D(x - other.x, y - other.y);
    }
    
    // Overload * for scalar multiplication
    Vector2D operator*(double scalar) const {
        return Vector2D(x * scalar, y * scalar);
    }
    
    // Overload == for comparison
    bool operator==(const Vector2D& other) const {
        return (x == other.x) && (y == other.y);
    }
    
    // Overload << for output (friend function)
    friend ostream& operator<<(ostream& os, const Vector2D& v) {
        os << "(" << v.x << ", " << v.y << ")";
        return os;
    }
};

int main() {
    Vector2D v1(3, 4);
    Vector2D v2(1, 2);
    
    cout << "v1 = " << v1 << endl;
    cout << "v2 = " << v2 << endl;
    cout << "v1 + v2 = " << (v1 + v2) << endl;
    cout << "v1 - v2 = " << (v1 - v2) << endl;
    cout << "v1 * 2 = " << (v1 * 2) << endl;
    cout << "v1 == v2: " << (v1 == v2 ? "true" : "false") << endl;
    
    return 0;
}

Runtime: Virtual Functions & Dynamic Dispatch

Virtual functions enable the "magic" of runtime polymorphism. When you call a virtual function through a base class pointer, C++ looks up the correct function in the vtable at runtime.

virtual_functions.cpp 
#include <iostream>
using namespace std;

class Animal {
protected:
    string name;

public:
    Animal(string n) : name(n) {}
    
    // Virtual function - enables runtime polymorphism
    virtual void speak() {
        cout << name << " makes a generic sound" << endl;
    }
    
    virtual void move() {
        cout << name << " moves somehow" << endl;
    }
    
    // CRITICAL: Virtual destructor for proper cleanup!
    virtual ~Animal() {
        cout << "~Animal: " << name << endl;
    }
};

class Dog : public Animal {
public:
    Dog(string n) : Animal(n) {}
    
    // Override - replaces base implementation
    void speak() override {
        cout << name << " barks: Woof! Woof!" << endl;
    }
    
    void move() override {
        cout << name << " runs on four legs" << endl;
    }
    
    ~Dog() { cout << "~Dog" << endl; }
};

class Bird : public Animal {
public:
    Bird(string n) : Animal(n) {}
    
    void speak() override {
        cout << name << " chirps: Tweet! Tweet!" << endl;
    }
    
    void move() override {
        cout << name << " flies through the air" << endl;
    }
    
    ~Bird() { cout << "~Bird" << endl; }
};

// This function works with ANY Animal!
void makeAnimalAct(Animal* animal) {
    animal->speak();  // Calls correct version at RUNTIME
    animal->move();
}

int main() {
    cout << "=== Runtime Polymorphism Demo ===" << endl;
    
    // Array of base class pointers to derived objects
    Animal* animals[3];
    animals[0] = new Animal("Generic");
    animals[1] = new Dog("Buddy");
    animals[2] = new Bird("Tweety");
    
    // Same code, different behavior!
    for (int i = 0; i < 3; i++) {
        cout << "\n--- Animal " << i+1 << " ---" << endl;
        makeAnimalAct(animals[i]);
    }
    
    cout << "\n=== Cleanup ===" << endl;
    for (int i = 0; i < 3; i++) {
        delete animals[i];
    }
    
    return 0;
}
‚ö†Ô∏è Always Use Virtual Destructors!

When deleting a derived object through a base class pointer, only the virtual destructor ensures proper cleanup. Without it, the derived class destructor won't be called, causing memory leaks!

Pure Virtual Functions & Abstract Classes

A pure virtual function (= 0) has no implementation in the base class. Classes with at least one pure virtual function are abstract and cannot be instantiated.

Regular Virtual

virtual void draw() { ... }
‚Üì
Has default implementation
‚Üì
Override is optional
VS

Pure Virtual

virtual void draw() = 0;
‚Üì
No implementation
‚Üì
Override is REQUIRED
abstract_class.cpp 
#include <iostream>
using namespace std;

// Abstract class - defines an interface
class Shape {
protected:
    string color;

public:
    Shape(string c) : color(c) {}
    
    // Pure virtual functions - MUST be implemented by derived classes
    virtual double area() = 0;
    virtual double perimeter() = 0;
    virtual void draw() = 0;
    
    // Regular virtual function with default implementation
    virtual void describe() {
        cout << "A " << color << " shape" << endl;
    }
    
    virtual ~Shape() {}
};

class Circle : public Shape {
private:
    double radius;

public:
    Circle(string c, double r) : Shape(c), radius(r) {}
    
    // Must implement all pure virtual functions
    double area() override { return 3.14159 * radius * radius; }
    double perimeter() override { return 2 * 3.14159 * radius; }
    
    void draw() override {
        cout << "Drawing " << color << " circle (r=" << radius << ")" << endl;
    }
};

class Rectangle : public Shape {
private:
    double width, height;

public:
    Rectangle(string c, double w, double h) : Shape(c), width(w), height(h) {}
    
    double area() override { return width * height; }
    double perimeter() override { return 2 * (width + height); }
    
    void draw() override {
        cout << "Drawing " << color << " rectangle (" << width << "x" << height << ")" << endl;
    }
};

int main() {
    // Shape s("Red");  // ERROR! Cannot instantiate abstract class
    
    Shape* shapes[2];
    shapes[0] = new Circle("Blue", 5);
    shapes[1] = new Rectangle("Green", 4, 6);
    
    for (int i = 0; i < 2; i++) {
        shapes[i]->draw();
        shapes[i]->describe();
        cout << "  Area: " << shapes[i]->area() << endl;
        cout << "  Perimeter: " << shapes[i]->perimeter() << endl;
        cout << endl;
        delete shapes[i];
    }
    
    return 0;
}
üí° Key Polymorphism Takeaways

‚Ä¢ Use function overloading for compile-time polymorphism with different parameter types
‚Ä¢ Use virtual functions for runtime polymorphism with inheritance hierarchies
‚Ä¢ Always use override keyword when overriding virtual functions
‚Ä¢ Always declare destructors as virtual in base classes
‚Ä¢ Use abstract classes to define interfaces and enforce implementation

Section 02

Types of Constructors

Constructors are special member functions that initialize objects when they are created

üî®

Default Constructor

Takes no arguments. Called automatically when an object is created without arguments.

üìù

Parameterized Constructor

Takes parameters to initialize object with specific values.

üìã

Copy Constructor

Creates a new object as a copy of an existing object.

üöÄ

Move Constructor (C++11)

Transfers resources from a temporary object efficiently.

üéØ

Delegating Constructor

One constructor calls another constructor of the same class.

‚ö°

Explicit Constructor

Prevents implicit type conversions with explicit keyword.

All Constructor Types in Action

constructors.cpp 
#include <iostream>
#include <cstring>
using namespace std;

class MyString {
private:
    char* data;
    int length;

public:
    // 1. Default Constructor
    MyString() : data(nullptr), length(0) {
        cout << "Default constructor" << endl;
    }
    
    // 2. Parameterized Constructor
    MyString(const char* str) {
        cout << "Parameterized constructor" << endl;
        length = strlen(str);
        data = new char[length + 1];
        strcpy(data, str);
    }
    
    // 3. Copy Constructor
    MyString(const MyString& other) {
        cout << "Copy constructor" << endl;
        length = other.length;
        data = new char[length + 1];
        strcpy(data, other.data);
    }
    
    // 4. Move Constructor (C++11)
    MyString(MyString&& other) noexcept {
        cout << "Move constructor" << endl;
        data = other.data;
        length = other.length;
        other.data = nullptr;
        other.length = 0;
    }
    
    // Destructor
    ~MyString() {
        cout << "Destructor" << endl;
        delete[] data;
    }
    
    void print() const {
        if (data) cout << "String: \"" << data << "\" (len: " << length << ")" << endl;
        else cout << "String: empty" << endl;
    }
};

MyString createString() {
    return MyString("Temporary");
}

int main() {
    cout << "--- s1 (default) ---" << endl;
    MyString s1;
    s1.print();
    
    cout << "\n--- s2 (parameterized) ---" << endl;
    MyString s2("Hello World");
    s2.print();
    
    cout << "\n--- s3 (copy of s2) ---" << endl;
    MyString s3 = s2;
    s3.print();
    
    cout << "\n--- s4 (move) ---" << endl;
    MyString s4 = createString();
    s4.print();
    
    cout << "\n--- End of main ---" << endl;
    return 0;
}
‚ö†Ô∏è Rule of Three / Five

If you define any of: destructor, copy constructor, or copy assignment ‚Äî define all three. In C++11+, also consider move constructor and move assignment (Rule of Five).

Initializer Lists & Delegating Constructors

initializer_list.cpp 
#include <iostream>
using namespace std;

class Person {
private:
    const int id;         // const - MUST use initializer list
    string name;
    int age;

public:
    // Initializer list (required for const members)
    Person(int i, string n, int a) : id(i), name(n), age(a) {
        cout << "Created Person: " << name << endl;
    }
    
    void display() const {
        cout << "ID: " << id << ", Name: " << name << ", Age: " << age << endl;
    }
};

class Rectangle {
private:
    double width, height;

public:
    // Delegating constructor - calls another constructor
    Rectangle() : Rectangle(1.0, 1.0) {
        cout << "Default delegated to parameterized" << endl;
    }
    
    Rectangle(double w, double h) : width(w), height(h) {
        cout << "Parameterized: " << w << "x" << h << endl;
    }
    
    // Explicit - prevents implicit conversion
    explicit Rectangle(double side) : width(side), height(side) {
        cout << "Explicit square: " << side << endl;
    }
    
    void print() const {
        cout << "Rectangle: " << width << " x " << height << endl;
    }
};

int main() {
    Person p(101, "Alice", 25);
    p.display();
    
    cout << endl;
    Rectangle r1;           // Delegates
    r1.print();
    
    Rectangle r2(5.0, 3.0);
    r2.print();
    
    Rectangle r3(4.0);       // Explicit square
    r3.print();
    
    // Rectangle r4 = 5.0;   // ERROR! explicit prevents this
    
    return 0;
}
Section 03

Inline Functions

Inline functions suggest the compiler replace function calls with the actual code for performance

‚ö°

Performance Benefit

Eliminates function call overhead. Best for small, frequently called functions.

üìè

Code Size Trade-off

Inlining increases code size as the function body is duplicated at each call site.

ü§ñ

Compiler Discretion

The inline keyword is a suggestion. The compiler decides what to actually inline.

Inline Functions in Practice

inline_functions.cpp 
#include <iostream>
using namespace std;

// Inline functions - suggest inlining to compiler
inline int square(int x) { return x * x; }
inline int max(int a, int b) { return (a > b) ? a : b; }
inline double celsiusToF(double c) { return c * 9.0/5.0 + 32; }

class Circle {
private:
    double radius;

public:
    Circle(double r) : radius(r) {}
    
    // Methods defined inside class are implicitly inline
    double getRadius() const { return radius; }
    double area() const { return 3.14159 * radius * radius; }
    double circumference() const { return 2 * 3.14159 * radius; }
};

int main() {
    cout << "square(5) = " << square(5) << endl;
    cout << "max(10, 7) = " << max(10, 7) << endl;
    cout << "100¬∞C = " << celsiusToF(100) << "¬∞F" << endl;
    
    Circle c(5.0);
    cout << "\nCircle (r=" << c.getRadius() << "):" << endl;
    cout << "  Area: " << c.area() << endl;
    cout << "  Circumference: " << c.circumference() << endl;
    
    return 0;
}
üí° When to Use Inline

Good: Small functions (1-5 lines), getters/setters, simple calculations
Avoid: Functions with loops, recursion, complex logic, or large bodies

Section 04

Static Members

Static members belong to the class itself, not to individual objects

üìä

Static Variables

Shared by all instances. Only one copy exists regardless of object count.

üîß

Static Methods

Called without an object. Cannot access non-static members or this.

üè≠

Use Cases

Object counters, factory methods, utility functions, shared configuration.

Static Members Example

static_members.cpp 
#include <iostream>
using namespace std;

class BankAccount {
private:
    string owner;
    double balance;
    int accountId;
    
    // Static members - shared by ALL instances
    static int totalAccounts;
    static double interestRate;
    static int nextId;

public:
    BankAccount(string name, double initial) 
        : owner(name), balance(initial), accountId(nextId++) {
        totalAccounts++;
        cout << "Account #" << accountId << " created" << endl;
    }
    
    ~BankAccount() { totalAccounts--; }
    
    // Static methods - called without an object
    static int getTotalAccounts() { return totalAccounts; }
    static double getInterestRate() { return interestRate; }
    
    static void setInterestRate(double rate) {
        if (rate >= 0 && rate <= 0.20) interestRate = rate;
    }
    
    void applyInterest() { balance += balance * interestRate; }
    
    void display() const {
        cout << "#" << accountId << ": " << owner << " - $" << balance << endl;
    }
};

// Static member initialization (required outside class)
int BankAccount::totalAccounts = 0;
double BankAccount::interestRate = 0.05;
int BankAccount::nextId = 1001;

int main() {
    cout << "Total: " << BankAccount::getTotalAccounts() << endl;
    
    BankAccount acc1("Alice", 1000);
    BankAccount acc2("Bob", 2000);
    
    cout << "Total: " << BankAccount::getTotalAccounts() << endl;
    
    BankAccount::setInterestRate(0.07);
    acc1.applyInterest();
    acc2.applyInterest();
    
    acc1.display();
    acc2.display();
    
    return 0;
}
üìò Static Member Rules

‚Ä¢ Static data members must be defined outside the class
‚Ä¢ Static methods cannot access this or non-static members
‚Ä¢ Call with ClassName::method()

Section 05

Nested Classes

Classes defined inside other classes for logical grouping and encapsulation

ü™Ü

What are Nested Classes?

A class defined within another class. Useful for helper classes only relevant to the outer class.

üîê

Access Control

Nested classes can access private members of the enclosing class.

üì¶

Use Cases

Iterator classes, node classes for data structures, builder patterns.

Linked List with Nested Node & Iterator

linked_list.cpp 
#include <iostream>
using namespace std;

class LinkedList {
private:
    // Private nested Node class
    struct Node {
        int data;
        Node* next;
        Node(int d) : data(d), next(nullptr) {}
    };
    
    Node* head;
    int size;

public:
    // Public nested Iterator class
    class Iterator {
    private:
        Node* current;
    public:
        Iterator(Node* n) : current(n) {}
        int& operator*() { return current->data; }
        Iterator& operator++() { current = current->next; return *this; }
        bool operator!=(const Iterator& o) const { return current != o.current; }
    };
    
    LinkedList() : head(nullptr), size(0) {}
    
    ~LinkedList() {
        while (head) {
            Node* temp = head;
            head = head->next;
            delete temp;
        }
    }
    
    void pushBack(int val) {
        Node* newNode = new Node(val);
        if (!head) { head = newNode; }
        else {
            Node* t = head;
            while (t->next) t = t->next;
            t->next = newNode;
        }
        size++;
    }
    
    Iterator begin() { return Iterator(head); }
    Iterator end() { return Iterator(nullptr); }
    int getSize() const { return size; }
};

int main() {
    LinkedList list;
    list.pushBack(10);
    list.pushBack(20);
    list.pushBack(30);
    
    cout << "Size: " << list.getSize() << endl;
    
    // Using nested Iterator
    cout << "Elements: ";
    for (LinkedList::Iterator it = list.begin(); it != list.end(); ++it) {
        cout << *it << " ";
    }
    cout << endl;
    
    // Range-based for loop works too!
    cout << "Range-for: ";
    for (int val : list) {
        cout << val << " ";
    }
    cout << endl;
    
    return 0;
}

‚Ä¢ Helper classes only relevant to outer class (Node in List)

‚Ä¢ Implementation details to hide
