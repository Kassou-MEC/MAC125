Section 01

Understanding Structures

Structures allow you to group related variables of different types under a single name

üì¶

What is a Struct?

A structure is a user-defined data type that groups related variables (called members) together. Each member can have a different data type.

üîì

Public by Default

All members in a struct are public by default, meaning they can be accessed directly from outside the structure.

üéØ

Use Cases

Ideal for simple data containers like points, rectangles, or records where you just need to group data without complex behavior.

Basic Structure Definition

Define a structure using the struct keyword, then create instances (objects) of that type:

basic_struct.cpp 
#include <iostream>
using namespace std;

// Define a structure
struct Point {
    double x;
    double y;
};

// Structure with more members
struct Student {
    string name;
    int age;
    double gpa;
    string major;
};

int main() {
    // Create and initialize a Point
    Point p1;
    p1.x = 3.5;
    p1.y = 7.2;
    
    // Aggregate initialization
    Point p2 = {1.0, 2.0};
    
    // Create a Student
    Student s1 = {"Alice", 20, 3.85, "Computer Science"};
    
    // Access members with dot operator
    cout << "Point 1: (" << p1.x << ", " << p1.y << ")" << endl;
    cout << "Point 2: (" << p2.x << ", " << p2.y << ")" << endl;
    cout << "\nStudent: " << s1.name << endl;
    cout << "Age: " << s1.age << endl;
    cout << "GPA: " << s1.gpa << endl;
    cout << "Major: " << s1.major << endl;
    
    return 0;
}

Structures with Functions

Structures can be passed to functions and can also contain member functions:

struct_functions.cpp 
#include <iostream>
#include <cmath>
using namespace std;

struct Rectangle {
    double width;
    double height;
    
    // Member function inside struct
    double area() {
        return width * height;
    }
    
    double perimeter() {
        return 2 * (width + height);
    }
};

// Function that takes a struct by value
void printRect(Rectangle r) {
    cout << "Width: " << r.width << ", Height: " << r.height << endl;
}

// Function that takes a struct by reference (efficient!)
void scale(Rectangle &r, double factor) {
    r.width *= factor;
    r.height *= factor;
}

int main() {
    Rectangle rect = {10.0, 5.0};
    
    cout << "Original rectangle:" << endl;
    printRect(rect);
    cout << "Area: " << rect.area() << endl;
    cout << "Perimeter: " << rect.perimeter() << endl;
    
    // Scale the rectangle
    scale(rect, 2.0);
    
    cout << "\nAfter scaling by 2:" << endl;
    printRect(rect);
    cout << "Area: " << rect.area() << endl;
    
    return 0;
}

Section 02

Introduction to Classes

Classes are the foundation of Object-Oriented Programming, combining data and behavior with access control

struct

  • Members are public by default
  • Simple data containers
  • C-compatible
  • Good for POD (Plain Old Data)
  • Typically no complex behavior

class

  • Members are private by default
  • Encapsulates data + behavior
  • Full OOP support
  • Constructors/Destructors
  • Access control (private/public/protected)

Basic Class Definition

A class combines data (attributes) and functions (methods) with access control:

basic_class.cpp 
#include <iostream>
using namespace std;

class Circle {
private:
    double radius;  // Private member - only accessible inside class

public:
    // Constructor - initializes the object
    Circle(double r) {
        radius = r;
    }
    
    // Getter - provides read access to private member
    double getRadius() {
        return radius;
    }
    
    // Setter - provides write access with validation
    void setRadius(double r) {
        if (r > 0) {
            radius = r;
        }
    }
    
    // Member functions (methods)
    double area() {
        return 3.14159 * radius * radius;
    }
    
    double circumference() {
        return 2 * 3.14159 * radius;
    }
    
    void display() {
        cout << "Circle with radius: " << radius << endl;
        cout << "Area: " << area() << endl;
        cout << "Circumference: " << circumference() << endl;
    }
};

int main() {
    // Create Circle objects
    Circle c1(5.0);
    Circle c2(3.0);
    
    cout << "=== Circle 1 ===" << endl;
    c1.display();
    
    cout << "\n=== Circle 2 ===" << endl;
    c2.display();
    
    // Modify using setter
    c2.setRadius(7.5);
    cout << "\n=== Circle 2 (modified) ===" << endl;
    c2.display();
    
    return 0;
}

Constructors and Destructors

Constructors initialize objects; destructors clean up when objects are destroyed:

constructors.cpp 
#include <iostream>
using namespace std;

class Book {
private:
    string title;
    string author;
    int pages;

public:
    // Default constructor
    Book() {
        title = "Unknown";
        author = "Unknown";
        pages = 0;
        cout << "Default constructor called" << endl;
    }
    
    // Parameterized constructor
    Book(string t, string a, int p) {
        title = t;
        author = a;
        pages = p;
        cout << "Parameterized constructor called for: " << title << endl;
    }
    
    // Constructor with initializer list (more efficient!)
    Book(string t, string a) : title(t), author(a), pages(100) {
        cout << "Initializer list constructor for: " << title << endl;
    }
    
    // Destructor - called when object is destroyed
    ~Book() {
        cout << "Destructor called for: " << title << endl;
    }
    
    void display() {
        cout << "\"" << title << "\" by " << author 
             << " (" << pages << " pages)" << endl;
    }
};

int main() {
    cout << "Creating books..." << endl;
    
    Book b1;  // Default constructor
    Book b2("1984", "George Orwell", 328);  // Parameterized
    Book b3("Dune", "Frank Herbert");  // Initializer list
    
    cout << "\nBook information:" << endl;
    b1.display();
    b2.display();
    b3.display();
    
    cout << "\nEnd of main - destructors will be called..." << endl;
    return 0;
}
üí° Best Practice

Use initializer lists in constructors for better performance, especially with complex objects. They initialize members directly rather than default-constructing then assigning.

Section 03

The Power of Encapsulation

Encapsulation bundles data with methods that operate on it, while restricting direct access to the internal state

üîí

Data Hiding

Private members can't be accessed directly from outside the class, protecting the internal state from invalid modifications.

‚úÖ

Validation

Setters can validate data before assignment, ensuring the object always remains in a valid state.

üîÑ

Flexibility

Internal implementation can change without affecting code that uses the class, as long as the public interface stays the same.

Access Specifier Within Class Derived Class Outside Class
public ‚úÖ Yes ‚úÖ Yes ‚úÖ Yes
protected ‚úÖ Yes ‚úÖ Yes ‚ùå No
private ‚úÖ Yes ‚ùå No ‚ùå No

Encapsulation in Practice

A well-encapsulated class protects its data and provides controlled access:

bank_account.cpp 
#include <iostream>
using namespace std;

class BankAccount {
private:
    string accountHolder;
    string accountNumber;
    double balance;
    int transactionCount;

public:
    // Constructor
    BankAccount(string holder, string accNum, double initial)
        : accountHolder(holder), accountNumber(accNum), 
          balance(initial), transactionCount(0) {}
    
    // Getters - provide read-only access
    string getHolder() const { return accountHolder; }
    double getBalance() const { return balance; }
    int getTransactions() const { return transactionCount; }
    
    // Deposit with validation
    bool deposit(double amount) {
        if (amount <= 0) {
            cout << "Error: Invalid deposit amount" << endl;
            return false;
        }
        balance += amount;
        transactionCount++;
        cout << "Deposited: $" << amount << endl;
        return true;
    }
    
    // Withdraw with validation
    bool withdraw(double amount) {
        if (amount <= 0) {
            cout << "Error: Invalid withdrawal amount" << endl;
            return false;
        }
        if (amount > balance) {
            cout << "Error: Insufficient funds" << endl;
            return false;
        }
        balance -= amount;
        transactionCount++;
        cout << "Withdrew: $" << amount << endl;
        return true;
    }
    
    void displayInfo() const {
        cout << "Account Holder: " << accountHolder << endl;
        cout << "Account Number: " << accountNumber << endl;
        cout << "Balance: $" << balance << endl;
        cout << "Transactions: " << transactionCount << endl;
    }
};

int main() {
    BankAccount account("Alice Johnson", "ACC-12345", 1000.0);
    
    cout << "=== Initial State ===" << endl;
    account.displayInfo();
    
    cout << "\n=== Transactions ===" << endl;
    account.deposit(500);
    account.withdraw(200);
    account.withdraw(5000);  // Should fail
    account.deposit(-100);   // Should fail
    
    cout << "\n=== Final State ===" << endl;
    account.displayInfo();
    
    return 0;
}
üìò The const Keyword

Methods marked const promise not to modify the object's state. This allows them to be called on const objects and helps catch bugs at compile time.

Understanding Inheritance

Inheritance allows a class to inherit properties and behaviors from another class, promoting code reuse

üëÜ

Base Class (Parent)

The class being inherited from. Contains common attributes and methods that will be shared with derived classes.

üëá

Derived Class (Child)

The class that inherits. Gets all public/protected members from the base class and can add its own unique members.

‚ôªÔ∏è

Code Reuse

Write common functionality once in the base class. All derived classes automatically get this functionality.

Basic Inheritance Syntax

Use a colon followed by the access specifier and base class name:

inheritance_basics.cpp 
#include <iostream>
using namespace std;

// Base class
class Animal {
protected:
    string name;
    int age;

public:
    Animal(string n, int a) : name(n), age(a) {
        cout << "Animal constructor called" << endl;
    }
    
    void eat() {
        cout << name << " is eating." << endl;
    }
    
    void sleep() {
        cout << name << " is sleeping." << endl;
    }
    
    void displayInfo() {
        cout << "Name: " << name << ", Age: " << age << endl;
    }
};

// Derived class - inherits from Animal
class Dog : public Animal {
private:
    string breed;

public:
    // Constructor calls base class constructor
    Dog(string n, int a, string b) : Animal(n, a), breed(b) {
        cout << "Dog constructor called" << endl;
    }
    
    // Dog-specific methods
    void bark() {
        cout << name << " says: Woof! Woof!" << endl;
    }
    
    void fetch() {
        cout << name << " is fetching the ball!" << endl;
    }
    
    void showBreed() {
        cout << name << " is a " << breed << endl;
    }
};

// Another derived class
class Cat : public Animal {
private:
    bool isIndoor;

public:
    Cat(string n, int a, bool indoor) : Animal(n, a), isIndoor(indoor) {
        cout << "Cat constructor called" << endl;
    }
    
    void meow() {
        cout << name << " says: Meow!" << endl;
    }
    
    void climb() {
        cout << name << " is climbing!" << endl;
    }
};

int main() {
    cout << "=== Creating a Dog ===" << endl;
    Dog dog("Buddy", 3, "Golden Retriever");
    
    cout << "\n=== Dog Actions ===" << endl;
    dog.displayInfo();  // Inherited from Animal
    dog.showBreed();     // Dog-specific
    dog.eat();           // Inherited from Animal
    dog.bark();          // Dog-specific
    dog.fetch();         // Dog-specific
    
    cout << "\n=== Creating a Cat ===" << endl;
    Cat cat("Whiskers", 5, true);
    
    cout << "\n=== Cat Actions ===" << endl;
    cat.displayInfo();   // Inherited from Animal
    cat.sleep();         // Inherited from Animal
    cat.meow();          // Cat-specific
    cat.climb();         // Cat-specific
    
    return 0;
}

üîë Key Point

Public inheritance is most common and represents an "is-a" relationship (a Dog is an Animal). Protected and private inheritance are rarer and represent "is-implemented-in-terms-of" relationships.

Method Overriding

Derived classes can override base class methods to provide specialized behavior:

method_overriding.cpp 
#include <iostream>
using namespace std;

class Shape {
protected:
    string name;

public:
    Shape(string n) : name(n) {}
    
    void describe() {
        cout << "This is a " << name << endl;
    }
    
    // Base implementation - will be overridden
    double area() {
        cout << "Area not defined for generic shape" << endl;
        return 0;
    }
};

class Rectangle : public Shape {
private:
    double width, height;

public:
    Rectangle(double w, double h) : Shape("Rectangle"), width(w), height(h) {}
    
    // Override area method
    double area() {
        return width * height;
    }
};

class Circle : public Shape {
private:
    double radius;

public:
    Circle(double r) : Shape("Circle"), radius(r) {}
    
    // Override area method
    double area() {
        return 3.14159 * radius * radius;
    }
};

class Triangle : public Shape {
private:
    double base, height;

public:
    Triangle(double b, double h) : Shape("Triangle"), base(b), height(h) {}
    
    // Override area method
    double area() {
        return 0.5 * base * height;
    }
};

int main() {
    Rectangle rect(10, 5);
    Circle circle(7);
    Triangle tri(8, 6);
    
    rect.describe();
    cout << "Area: " << rect.area() << endl;
    
    cout << endl;
    circle.describe();
    cout << "Area: " << circle.area() << endl;
    
    cout << endl;
    tri.describe();
    cout << "Area: " << tri.area() << endl;
    
    return 0;
}
Section 05

Introduction to Polymorphism

Polymorphism allows objects of different classes to be treated through a common interface

üé≠

Virtual Functions

Functions declared with virtual enable runtime polymorphism. The correct version is called based on the actual object type.

üìç

Base Pointers

A pointer to a base class can point to any derived class object, enabling flexible and extensible code.

üîÄ

Dynamic Binding

The decision of which function to call is made at runtime, not compile time, based on the actual object type.

Virtual Functions and Polymorphism

Virtual functions enable true polymorphic behavior with base class pointers:

polymorphism.cpp 
#include <iostream>
using namespace std;

class Shape {
protected:
    string name;

public:
    Shape(string n) : name(n) {}
    
    // Virtual function - enables polymorphism!
    virtual double area() {
        return 0;
    }
    
    virtual void draw() {
        cout << "Drawing a " << name << endl;
    }
    
    string getName() { return name; }
    
    // Virtual destructor - important for polymorphism!
    virtual ~Shape() {
        cout << "Shape destructor for " << name << endl;
    }
};

class Rectangle : public Shape {
private:
    double width, height;

public:
    Rectangle(double w, double h) : Shape("Rectangle"), width(w), height(h) {}
    
    // Override with 'override' keyword (C++11)
    double area() override {
        return width * height;
    }
    
    void draw() override {
        cout << "Drawing rectangle: " << width << " x " << height << endl;
    }
    
    ~Rectangle() {
        cout << "Rectangle destructor" << endl;
    }
};

class Circle : public Shape {
private:
    double radius;

public:
    Circle(double r) : Shape("Circle"), radius(r) {}
    
    double area() override {
        return 3.14159 * radius * radius;
    }
    
    void draw() override {
        cout << "Drawing circle with radius: " << radius << endl;
    }
    
    ~Circle() {
        cout << "Circle destructor" << endl;
    }
};

// Function that works with ANY shape!
void printShapeInfo(Shape* shape) {
    cout << "Shape: " << shape->getName() << endl;
    shape->draw();
    cout << "Area: " << shape->area() << endl;
    cout << endl;
}

int main() {
    // Create shapes
    Shape* shapes[3];
    shapes[0] = new Rectangle(10, 5);
    shapes[1] = new Circle(7);
    shapes[2] = new Rectangle(3, 4);
    
    cout << "=== Polymorphism in Action ===" << endl;
    
    // Same function call, different behavior!
    for (int i = 0; i < 3; i++) {
        printShapeInfo(shapes[i]);
    }
    
    // Calculate total area
    double totalArea = 0;
    for (int i = 0; i < 3; i++) {
        totalArea += shapes[i]->area();
    }
    cout << "Total area of all shapes: " << totalArea << endl;
    
    // Clean up
    cout << "\n=== Cleaning up ===" << endl;
    for (int i = 0; i < 3; i++) {
        delete shapes[i];
    }
    
    return 0;
}
‚ö†Ô∏è Virtual Destructor

Always declare the destructor as virtual in base classes when using polymorphism. Without it, derived class destructors won't be called when deleting through a base pointer, causing memory leaks!

Pure Virtual Functions and Abstract Classes

Abstract classes define interfaces that derived classes must implement:

abstract_class.cpp 
#include <iostream>
using namespace std;

// Abstract class - cannot be instantiated
class Vehicle {
protected:
    string brand;
    int year;

public:
    Vehicle(string b, int y) : brand(b), year(y) {}
    
    // Pure virtual functions - MUST be implemented by derived classes
    virtual void start() = 0;
    virtual void stop() = 0;
    virtual int getWheels() = 0;
    
    // Non-pure virtual - has default implementation
    virtual void honk() {
        cout << "Beep beep!" << endl;
    }
    
    void showInfo() {
        cout << year << " " << brand << endl;
    }
    
    virtual ~Vehicle() {}
};

class Car : public Vehicle {
public:
    Car(string b, int y) : Vehicle(b, y) {}
    
    void start() override {
        cout << brand << " car: Engine starting... Vroom!" << endl;
    }
    
    void stop() override {
        cout << brand << " car: Braking... Stopped." << endl;
    }
    
    int getWheels() override { return 4; }
};

class Motorcycle : public Vehicle {
public:
    Motorcycle(string b, int y) : Vehicle(b, y) {}
    
    void start() override {
        cout << brand << " motorcycle: Kick start... Vrrrm!" << endl;
    }
    
    void stop() override {
        cout << brand << " motorcycle: Stopping..." << endl;
    }
    
    int getWheels() override { return 2; }
    
    void honk() override {
        cout << "Honk honk!" << endl;
    }
};

int main() {
    // Vehicle v("Generic", 2020);  // ERROR! Cannot instantiate abstract class
    
    Car car("Toyota", 2022);
    Motorcycle bike("Harley", 2021);
    
    Vehicle* vehicles[] = {&car, &bike};
    
    for (Vehicle* v : vehicles) {
        v->showInfo();
        cout << "Wheels: " << v->getWheels() << endl;
        v->start();
        v->honk();
        v->stop();
        cout << endl;
    }
    
    return 0;
}
üìò Abstract vs Concrete Classes

A class with at least one pure virtual function (= 0) is abstract and cannot be instantiated. Derived classes must implement all pure virtual functions to become concrete (instantiable).
