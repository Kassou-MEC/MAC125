MODULE 01

Dynamic Arrays

Allocate arrays at runtime when size isn't known at compile time. Master memory management with new[] and delete[].

üìè

Runtime Sizing

Array size determined during execution, not compilation.

üß†

Heap Memory

Allocated on heap, persists until explicitly deleted.

üîÑ

Resizable

Can reallocate to grow or shrink as needed.

‚ö†Ô∏è

Manual Cleanup

Must use delete[] to prevent memory leaks.

Basic Dynamic Array

Create and manage arrays with runtime-determined sizes:

dynamic_array.cpp 
#include <iostream>
using namespace std;

int main() {
    int size;
    cout << "Enter array size: ";
    cin >> size;
    
    // Allocate dynamic array
    int* arr = new int[size];
    
    // Fill with values
    for (int i = 0; i < size; i++) {
        arr[i] = (i + 1) * 10;
    }
    
    // Display values
    cout << "Array: ";
    for (int i = 0; i < size; i++) {
        cout << arr[i] << " ";
    }
    cout << endl;
    
    // IMPORTANT: Free memory!
    delete[] arr;
    arr = nullptr;  // Good practice
    
    cout << "Memory freed successfully!" << endl;
    
    return 0;
}

2D Dynamic Arrays

Create dynamic matrices using arrays of pointers:

dynamic_2d.cpp 
#include <iostream>
using namespace std;

int main() {
    int rows = 3, cols = 4;
    
    // Allocate array of pointers (rows)
    int** matrix = new int*[rows];
    
    // Allocate each row
    for (int i = 0; i < rows; i++) {
        matrix[i] = new int[cols];
    }
    
    // Fill matrix
    int val = 1;
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            matrix[i][j] = val++;
        }
    }
    
    // Display matrix
    cout << "Matrix:" << endl;
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            cout << matrix[i][j] << "\t";
        }
        cout << endl;
    }
    
    // Delete in reverse order!
    for (int i = 0; i < rows; i++) {
        delete[] matrix[i];  // Delete each row
    }
    delete[] matrix;  // Delete array of pointers
    
    cout << "2D array freed!" << endl;
    
    return 0;
}
üö® Memory Leak Warning

Always delete in reverse order: inner arrays first, then the array of pointers. Missing any step causes memory leaks!

Resizing Dynamic Arrays

Grow arrays by allocating new memory and copying:

resize_array.cpp 
#include <iostream>
using namespace std;

// Function to resize array
int* resize(int* oldArr, int oldSize, int newSize) {
    int* newArr = new int[newSize];
    
    // Copy existing elements
    int copyCount = (oldSize < newSize) ? oldSize : newSize;
    for (int i = 0; i < copyCount; i++) {
        newArr[i] = oldArr[i];
    }
    
    // Initialize new elements to 0
    for (int i = oldSize; i < newSize; i++) {
        newArr[i] = 0;
    }
    
    delete[] oldArr;  // Free old memory
    return newArr;
}

int main() {
    int size = 3;
    int* arr = new int[size]{10, 20, 30};
    
    cout << "Original (size " << size << "): ";
    for (int i = 0; i < size; i++) cout << arr[i] << " ";
    cout << endl;
    
    // Resize to 6 elements
    arr = resize(arr, size, 6);
    size = 6;
    
    // Add new values
    arr[3] = 40; arr[4] = 50; arr[5] = 60;
    
    cout << "Resized (size " << size << "): ";
    for (int i = 0; i < size; i++) cout << arr[i] << " ";
    cout << endl;
    
    delete[] arr;
    return 0;
}
üí° Better Alternative

In modern C++, use std::vector for automatic resizing, memory management, and safety. Manual dynamic arrays are still important to understand for interviews and low-level work.

STATIC ARRAYS

int arr[100];

  • Size fixed at compile time
  • Stack allocated
  • Auto cleanup
  • Fast but inflexible
DYNAMIC ARRAYS

int* arr = new int[n];

  • Size set at runtime
  • Heap allocated
  • Manual delete[] required
  • Flexible, can resize

Creating & Using Namespaces

namespaces.cpp 
#include <iostream>

// Define namespace for math utilities
namespace MathUtils {
    const double PI = 3.14159265;
    
    double square(double x) {
        return x * x;
    }
    
    double circleArea(double r) {
        return PI * square(r);
    }
}

// Another namespace
namespace StringUtils {
    std::string toUpper(std::string s) {
        for (char& c : s) {
            if (c >= 'a' && c <= 'z') c -= 32;
        }
        return s;
    }
}

int main() {
    // Method 1: Full qualification
    std::cout << "PI = " << MathUtils::PI << std::endl;
    std::cout << "5¬≤ = " << MathUtils::square(5) << std::endl;
    
    // Method 2: using directive
    using namespace MathUtils;
    std::cout << "Circle area (r=3): " << circleArea(3) << std::endl;
    
    // Method 3: using declaration
    using StringUtils::toUpper;
    std::cout << toUpper("hello world") << std::endl;
    
    return 0;
}

namespace Company::Graphics::Shapes {

    public:
        double width, height;

    // Full path

    Shapes::Rectangle r2(10, 4);
    cout << "r2 area: " << r2.area() << endl;
    
    return 0;
}

Header File (calculator.h):

calculator.h 
#ifndef CALCULATOR_H
#define CALCULATOR_H

namespace Calc {
    class Calculator {
    private:
        double memory;
    
    public:
        Calculator();
        double add(double a, double b);
        double subtract(double a, double b);
        double multiply(double a, double b);
        double divide(double a, double b);
        void store(double val);
        double recall();
    };
}

#endif

Implementation File (calculator.cpp):

calculator.cpp 
#include "calculator.h"
#include <stdexcept>

namespace Calc {
    Calculator::Calculator() : memory(0) {}
    
    double Calculator::add(double a, double b) {
        return a + b;
    }
    
    double Calculator::subtract(double a, double b) {
        return a - b;
    }
    
    double Calculator::multiply(double a, double b) {
        return a * b;
    }
    
    double Calculator::divide(double a, double b) {
        if (b == 0) throw std::runtime_error("Division by zero!");
        return a / b;
    }
    
    void Calculator::store(double val) { memory = val; }
    double Calculator::recall() { return memory; }
}

Main File (main.cpp):

main.cpp 
#include <iostream>
#include "calculator.h"
using namespace std;

int main() {
    Calc::Calculator calc;
    
    cout << "10 + 5 = " << calc.add(10, 5) << endl;
    cout << "10 - 5 = " << calc.subtract(10, 5) << endl;
    cout << "10 * 5 = " << calc.multiply(10, 5) << endl;
    cout << "10 / 5 = " << calc.divide(10, 5) << endl;
    
    calc.store(42);
    cout << "Memory: " << calc.recall() << endl;
    
    return 0;
}

Writing to Files

write_file.cpp 
#include <iostream>
#include <fstream>
using namespace std;

int main() {
    // Create and open file
    ofstream outFile("students.txt");
    
    // Check if opened successfully
    if (!outFile) {
        cerr << "Error creating file!" << endl;
        return 1;
    }
    
    // Write data
    outFile << "Student Records" << endl;
    outFile << "===============" << endl;
    outFile << "Name: Alice, GPA: 3.8" << endl;
    outFile << "Name: Bob, GPA: 3.5" << endl;
    outFile << "Name: Carol, GPA: 3.9" << endl;
    
    outFile.close();
    cout << "File written successfully!" << endl;
    
    // Append mode
    ofstream appendFile("students.txt", ios::app);
    appendFile << "Name: David, GPA: 3.7" << endl;
    appendFile.close();
    cout << "Data appended!" << endl;
    
    return 0;
}

Reading from Files

read_file.cpp 
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

int main() {
    ifstream inFile("students.txt");
    
    if (!inFile) {
        cerr << "Error opening file!" << endl;
        return 1;
    }
    
    // Method 1: Read line by line
    cout << "=== Line by Line ===" << endl;
    string line;
    while (getline(inFile, line)) {
        cout << line << endl;
    }
    
    // Reset to beginning
    inFile.clear();
    inFile.seekg(0);
    
    // Method 2: Read word by word
    cout << "\n=== Word by Word ===" << endl;
    string word;
    int count = 0;
    while (inFile >> word) {
        count++;
    }
    cout << "Total words: " << count << endl;
    
    inFile.close();
    return 0;
}

CSV File Processing

csv_process.cpp 
#include <iostream>
#include <fstream>
#include <sstream>
#include <string>
#include <vector>
using namespace std;

struct Student {
    string name;
    int age;
    double gpa;
};

int main() {
    // Create CSV file
    ofstream csvOut("data.csv");
    csvOut << "Name,Age,GPA\n";
    csvOut << "Alice,20,3.8\n";
    csvOut << "Bob,22,3.5\n";
    csvOut << "Carol,19,3.9\n";
    csvOut.close();
    cout << "CSV created!" << endl;
    
    // Parse CSV file
    ifstream csvIn("data.csv");
    vector<Student> students;
    string line;
    
    getline(csvIn, line);  // Skip header
    
    while (getline(csvIn, line)) {
        stringstream ss(line);
        Student s;
        string temp;
        
        getline(ss, s.name, ',');
        getline(ss, temp, ',');
        s.age = stoi(temp);
        getline(ss, temp, ',');
        s.gpa = stod(temp);
        
        students.push_back(s);
    }
    csvIn.close();
    
    // Display parsed data
    cout << "\nParsed Students:" << endl;
    for (const auto& s : students) {
        cout << s.name << " | Age: " << s.age 
             << " | GPA: " << s.gpa << endl;
    }
    
    return 0;
}

Binary File I/O

binary_io.cpp 
#include <iostream>
#include <fstream>
using namespace std;

struct Record {
    int id;
    char name[50];
    double salary;
};

int main() {
    // Write binary data
    ofstream binOut("records.bin", ios::binary);
    
    Record r1 = {1, "Alice", 75000.50};
    Record r2 = {2, "Bob", 68000.25};
    
    binOut.write(reinterpret_cast<char*>(&r1), sizeof(Record));
    binOut.write(reinterpret_cast<char*>(&r2), sizeof(Record));
    binOut.close();
    cout << "Binary file written!" << endl;
    
    // Read binary data
    ifstream binIn("records.bin", ios::binary);
    Record temp;
    
    cout << "\nReading records:" << endl;
    while (binIn.read(reinterpret_cast<char*>(&temp), sizeof(Record))) {
        cout << "ID: " << temp.id 
             << ", Name: " << temp.name 
             << ", Salary: $" << temp.salary << endl;
    }
    binIn.close();
    
    return 0;
}
‚ö†Ô∏è Binary Portability

Binary files may not be portable across systems due to differences in byte order (endianness) and struct padding. Use text formats like JSON/XML for cross-platform data.
